# Define two example tuples
tuple1 = (4, 7, 2, 8, 10, 3)
tuple2 = (5, 9, 1, 6)

# i) Finding the maximum and minimum values in tuple1
max_value = max(tuple1)
min_value = min(tuple1)
print(f"Maximum value in tuple1: {max_value}")
print(f"Minimum value in tuple1: {min_value}")

# iii) Summing two tuples
sum_tuple = tuple1 + tuple2
print(f"Sum of tuple1 and tuple2: {sum_tuple}")

# iv) Duplicating a tuple
duplicated_tuple = tuple1 * 2
print(f"Duplicated tuple1: {duplicated_tuple}")

# v) Slicing a tuple
sliced_tuple = tuple1[1:4]
print(f"Sliced tuple1 (from index 1 to 3): {sliced_tuple}")

# vi) Converting a tuple to a list
tuple_to_list = list(tuple1)
print(f"List obtained from tuple1: {tuple_to_list}")
